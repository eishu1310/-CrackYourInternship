# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def removeElements(self, head, val):
        """
        :type head: ListNode
        :type val: int
        :rtype: ListNode
        """
        temp = head

        while temp and temp.val == val:
            temp = temp.next
        
        ret = temp
        prev = temp

        while temp:
            if temp.val != val:
                if temp != prev:
                    prev.next = temp
                prev = temp
            temp = temp.next
        if prev:
            prev.next = None
        return ret
